نمايشنامه سه خواهر به کارگردانی محمود رضا رحيمی در تالار نو در مهرماه ۱۳۸۴ بر روی صحنه رفته است. اين نمايشنامه اثر نمايشنامه و داستان نويس طنز پرداز روسی آنتون چخوف است که توسط اين کارگردان سال پيش به مناسبت هفته چخوف برروی صحنه رفت. 
سه خواهر به همراه مرغ دريايی، دايی وانيا و باغ آلبالو از نمايشنامه های بلند آنتون چخوف اند که نويسنده در طول ۸سال آخر عمرخود (۱۸۹۶- ۱۹۰۴) آنها را نوشت. خلق اين آثار مقام چخوف نمايشنامه نويس را هم تراز چخوف داستان نويس کرد.
اين نمايشنامه های بلند به خوبی شرايط تغيير وتحول روسيه را درابعاد گوناگون و بدون شعار زدگی ارائه می کند. اين تغيير وتحول در مرغ دريايی در چالش هنر و هنرمند به ضرورت ديدگاه جديد به زندگی اشاره می کند. 
در دايی وانيا تاکيد می کند که انسان های ساده در عصر جديد دوامی ندارند. در سه خواهر که زنانه ترين نمايش چخوف است، ياد آوری می کند که زنان بدون اتکا به خود نمی توانند به آرزوهای خود برسند و در باغ آلبالو، چخوف به تغيير وتحول اجتناب ناپذير گذر از دوران فئودالی به دوران بورژوازی اشاره می کند.
سه خواهر به نامه های الگا، ماشا و ايرنا که به همراه برادرشان آندره در خانه پدری زندگی می کنند، آرزوی رفتن به مسکو را دارند. آنها پدر و مادر خود را از دست داده اند. پدر آنها افسر عاليرتبه ارتش بوده که در شهر کوچکی همراه با خانواده اش زندگی می کرده است. آنها به روال زمان پدرشان در پی برگزاری ميهمانی های با شکوه هستند و هر کدام به دو سه زبان صحبت می کنند. آندره برادر آنها تمرين ويلون می کند و آنها دوست دارند که او استاد دانشگاه شود. آندره با زنی معمولی ازدواج می کند وارد شورای شهر می شود. به مرور ناتاشا زن آندره خانه را از چنگ آنها در می آورد. سرانجام سه خواهر به آرزوی خود نمی رسند.
محمود رضا رحيمی نمايش سه خواهر را بر مينای "پرفورمنس آرت" کارگردانی کرده است. کارگردان معتقد است که در عصر حاضر کلام تاثير سابق را در مخاطب ندارد. بنابراين تصاوير نقش گوياتری را برای انتقال مفاهيم بازی می کنند. 
"پرفورمنس آرت"
در بروشور نمايش توضيحی در خصوص "پرفورمنس آرت" آمده است: "وقتی هنرهای تجسمی با پيچيدگی های مفهومی باعث شد که هنرمند خود برای تکميل آن به ميدان هنری قدم گذارد. تماشاگر در حضور موثر و با ارتباطی که بين اثر هنری و هنرمند به وجود می آمد به درک بهتر اثر هنری و ايجاد ارتباط با آن پرداخت."
"از همين جا بود که هنرهای مفهومی پا به عرصه درام گذاشت و با نام "پرفورمنس آرت" که توسط منتقدان دهه های سی و چهل قرن بيست بر آن گذاشته شد، به راهی برای توليد آثار هنری تبديل شد. از اين جهت "پرفورمنس آرت" توانايی اجرای درام را صد چندان کرد و از آن پس ملزومات آن يعنی هنر بدن- هنر نقاشی- هنر مجسمه- هنر ويديو- هنر محيط - هنر رقص- هنر موسيقی و بدون اينکه از حيطه خلاقيت خودشان خارج شوند در هنرهای نمايشی اجرا به تبيين نکات تازه تری پرداختند."
به اين منظور تماشاچی از زمانی که از در ورودی قدم به داخل می گذارد تا رسيدن به صحنه نمايش اين "پرفورمنس آرت" بر مبنای نمايش سه خواهر برای او اجرا می شود. در ابتدا بازيگران ميمو درام ( نمايش بی کلام) نمايش به طور ثابت و به شکل های مختلف بدنی تماشاچی را به سالن اجرا راهنمايی می کنند.
در راهروی منتهی به سالن اجرا دو مجسمه قراردارد که از شخصيت های نمايش هستند ( توزنباخ و ماشا) . بازيگران در حال و هوای شخصيت بازی در کنار تماشاچيان هستند. در اين راهرو نمايشگاهی از عکس های اجرا نمايش در معرض ديد تماشاچی است. 
پس از مدت کوتاهی تماشاچی وارد سالن نمايش می شود و درام سه خواهر آغاز می شود. در ميان صحنه های نمايش سه بازيگری ميمو درام سراپا سياه پوش حضور پيدا می کنند و در ارتباط با صحنه مفهومی را اجرا می کنند. 
زمان اين نمايش ۷۵ دقيقه است. اصل نمايشنامه حدود سه ساعت زمان می برد. برای اجرا گروه نمايشی با حذف بخش عظيمی از نمايش و شخصيت ها و جايگزين کردن ميمودرام و مجسمه ها به نمايش خواسته است که با مفاهيم تصويری جای مفاهيم کلامی را بگيرد.
اجرای ميمودرام ونمايش مجسمه ها و عکس ها قبل از آغاز نمايش ذهن تماشاچی را به خود مشغول می کند. ولی وقفه کوتاهی که تا شروع نمايش اين حال وهوا از ذهن می رود. 
"پرفورمنس آرت" را نقطه اتصال هنر تجسمی وهنر نمايش می دانند. در اين نوع از هنر بيننده در جريان روند خلاقيت قرار می گيرد. از وپژگی های "پرفورمنس آرت" اين خط اتصال و اتفاقات در لحظه است. 
در سه خواهر وقتی تماشاچی وارد سالن نمايش می شود و وقفه کوتاه تا آغاز نمايش کاملاً اين ارتباط را قطع می کند. حضور قوی شخصيت نمايش سه خواهر، به ناچار بازيگران را به درون خود می کشاند، که با توجه به روش اجرا در مقابل آن قرارمی گرفت. 
استفاده از کلام به شکل صدای ضبط شده شخصيت های نمايشنامه های ديگر چخوف که در ميان توقف های اجرايی نمايش شنيده می شد. با توجه به اعتقاد بيشتر کارگردان به مفاهيم تصويری تا کلامی، در مقابل خواست او قرار می گرفت.
البته انتخاب گوشه هايی از متن های ديگر چخوف که يکپارچگی انديشه نويسنده را نشان می دهد، جالب توجه است. 
ضرورت تغيير دورانی به دوران ديگر از مفاهيم نمايشنامه های چخوف است. هرکس که شرايط تغيير را درک نکند، حتی دارای دانش وهنر باشد، محکوم به زوال است. انتخاب لباس های امروزی و استفاده از ابزارهای امروزی در نمايش، بيننده را به مفهوم درون نمايش نزديک می کرد.
